\section*{Glossary}
\addcontentsline{toc}{section}{Glossary}

\begin{description}[style=nextline, labelwidth=4cm, leftmargin=4.5cm]
  \item[AES-GCM] Advanced Encryption Standard -- Galois/Counter Mode. Authenticated encryption scheme.
  \item[CBOR] Concise Binary Object Representation. A binary data serialization format.
  \item[DHT] Distributed Hash Table. A decentralized key-value store.
  \item[HE] Homomorphic Encryption. Encryption supporting computation on ciphertexts.
  \item[MPC / SMPC] (Secure) Multi-Party Computation. Protocols enabling joint computation over private inputs without revealing them.
  \item[MP-SPDZ] Multi-Protocol SPDZ. A framework for multi-party computation supporting multiple protocol backends.
  \item[NAT] Network Address Translation.
  \item[P2P] Peer-to-Peer.
  \item[SPDZ] ``Speedz'' protocol for MPC with dishonest majority security.
  \item[SSS] Shamir's Secret Sharing. A threshold secret-sharing scheme.
  \item[Veilid] A privacy-first, open-source P2P application framework with onion routing and a secure DHT.
\end{description}
